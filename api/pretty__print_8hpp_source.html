32  for (std::size_t i = 0; i < arr.size(); ++i) {
33  os << new_indent;
34  pretty_print_impl<allocator_type, indent_size>(os, arr[i], new_indent);
35  if (i < arr.size() - 1) {

40  } else if (jv.is_object()) {
