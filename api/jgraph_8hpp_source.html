405  internal_id_type priv_get_vertex_internal_id(const std::string_view &vid) const {
406  auto hash = priv_hash_id(vid.data());
407 
408  for (std::size_t d = 0; d <= m_max_vid_distance; ++d) {
409  const auto vitr = m_vertex_id_table.find(hash);
410  if (vitr == m_vertex_id_table.end()) {
411  break;
412  }
413 
414  if (vitr->second == vid) {
415  return hash;
416  }
417  hash = (hash + 1) % k_max_internal_id;
418  }
419 
420  return k_max_internal_id; // Couldn't find
421  }
422 
423  internal_id_type priv_generate_vertex_internal_id(const std::string_view &vid) {
424  auto hash = priv_hash_id(vid.data());
425 
426  std::size_t distance = 0;
427  while (m_vertex_id_table.count(hash) > 0) {
428  hash = (hash + 1) % k_max_internal_id;
429  ++distance;
430  }
431  m_max_vid_distance = std::max(distance, m_max_vid_distance);
432 
433  m_vertex_id_table[hash] = vid.data();
434 
435  return hash;
436  }
437 
438  internal_id_type priv_generate_edge_id() {
439  return ++m_max_edge_id;
440  }

442  static internal_id_type priv_hash_id(const std::string_view &id) {
443  return metall::mtlldetail::MurmurHash64A(id.data(), id.length(), 1234);
444  }
