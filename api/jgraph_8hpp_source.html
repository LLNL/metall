147  bool register_vertex(const std::string_view &vertex_id) {
148  auto vpos = priv_locate_vertex(vertex_id);
149  if (vpos != m_vertex_directory.end()) {
150  return false; // Already exist
151  }
152 
153  m_vertex_storage.emplace_back(vertex_id,
154  typename key_value_pair_type::value_type{m_vertex_storage.get_allocator()});
155  m_vertex_directory.emplace(priv_hash_id(vertex_id),
156  vertex_directory_value_type{m_vertex_storage.size() - 1,
157  dst_vertex_directory_type{
158  m_vertex_directory.get_allocator()}});
159  return true;
160  }
161 
168  bool register_edge(const std::string_view &source_id, const std::string_view &destination_id,
169  const std::string_view &edge_id) {
170  register_vertex(source_id);
171  register_vertex(destination_id);
172 
173  m_edge_storage.emplace_back(edge_id, typename key_value_pair_type::value_type{m_vertex_storage.get_allocator()});
174  const auto edge_pos = m_edge_storage.size() - 1;
175  m_edge_directory.emplace(priv_hash_id(edge_id), edge_directory_value_type{edge_pos});
176 
177  auto src_pos = priv_locate_vertex(source_id);
178  auto dst_pos = priv_locate_vertex(destination_id);
179  vertex_directory_value_type &src_val = src_pos->second;
180  vertex_directory_value_type &dst_val = dst_pos->second;
181  src_val.dst_vertex_directory.emplace(priv_hash_id(destination_id),
182  dst_vertex_directory_value_type{dst_val.vertex_position, edge_pos});
183 
184  return true;
185  }

269  typename vertex_directory_type::const_iterator priv_locate_vertex(const std::string_view &vid) const {
270  const auto hash = priv_hash_id(vid.data());
271  auto range = m_vertex_directory.equal_range(hash);
272  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
273  const vertex_directory_value_type &value = itr->second;
274  if (m_vertex_storage[value.vertex_position].key() == vid) {
275  return itr; // Found the key
276  }
277  }
278  return m_vertex_directory.end(); // Couldn't find
279  }

281  typename vertex_directory_type::iterator priv_locate_vertex(const std::string_view &vid) {
282  const auto citr = const_cast<const jgraph *>(this)->priv_locate_vertex(vid);
283  return m_vertex_directory.erase(citr, citr); // Generate non-const iterator from const iterator
284  }

286  typename edge_directory_type::const_iterator priv_locate_edge(const std::string_view &edge_id) const {
287  const auto hash = priv_hash_id(edge_id);
288  auto range = m_edge_directory.equal_range(hash);
289  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
290  if (m_edge_storage[itr->second.edge_position].key() == edge_id) {
291  return itr; // Found the key
292  }
293  }
294  return m_edge_directory.end(); // Couldn't find
295  }
296 
297  typename edge_directory_type::iterator priv_locate_edge(const std::string_view &edge_id) {
298  const auto citr = const_cast<const jgraph *>(this)->priv_locate_edge(edge_id);
299  return m_edge_directory.erase(citr, citr); // Generate non-const iterator from const iterator
300  }
