103  bool insert(const key_type &key) {
104  const auto internal_id = priv_find_or_generate_internal_id(key);
105  if (m_unique && m_map.count(internal_id) == 1) {
106  return false;
107  }

109  // We delegate tuple to decide if value_type's constructor needs an allocator object.
110  // By taking this approach, we can avoid the complex logic of uses-allocator construction.

112  auto internal_value = internal_value_type{std::allocator_arg, m_map.get_allocator()};

114

286  internal_id_type priv_generate_internal_id(const key_type &key) {
287  auto internal_id = priv_hash_key(key, m_hash_seed);
288 
289  std::size_t distance = 0;
290  while (m_map.count(internal_id) > 0) {

293  }

296  return internal_id;

302  internal_id_type priv_find_internal_id(const key_type &key) const {
303  auto internal_id = priv_hash_key(key, m_hash_seed);
304 
305  for (std::size_t d = 0; d <= m_max_id_probe_distance; ++d) {
306  const auto itr = m_map.find(internal_id);
307  if (itr == m_map.end()) {
308  break;
309  }
310 
311  if (std::get<0>(itr->second) == key) {
312  return internal_id;
313  }
314  internal_id = priv_increment_internal_id(internal_id);
315  }
316 
317  return k_max_internal_id; // Couldn't find
318  }

323  internal_id_type priv_find_or_generate_internal_id(const key_type &key) {
324  auto internal_id = priv_find_internal_id(key);
325  if (internal_id == k_max_internal_id) { // Couldn't find
326  // Generate a new one.
327  internal_id = priv_generate_internal_id(key);
328  }
329  return internal_id;
330  }

332  static internal_id_type priv_hash_key(const key_type &key, [[maybe_unused]]const uint64_t seed) {
333 #ifdef METALL_CONTAINER_STRING_KEY_STORE_USE_SIMPLE_HASH
334  internal_id_type hash = key.empty() ? 0 : (uint8_t)key[0] % 2;
335 #else
336  auto hash = (internal_id_type)metall::mtlldetail::MurmurHash64A(key.data(), (int)key.length(), seed);
337 #endif
338  if (hash == k_max_internal_id) {
339  hash = priv_increment_internal_id(hash);
340  }
341  assert(hash != k_max_internal_id);
342  return hash;
343  }
344 
345  static internal_id_type priv_increment_internal_id(const internal_id_type id) {
346  const auto new_id = (id + 1) % k_max_internal_id;
347  assert(new_id != k_max_internal_id);
348  return new_id;
349  }
