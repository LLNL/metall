155  pointer priv_allocate(const size_type n) const {
156  if (priv_max_size() < n) {
157  throw std::bad_array_new_length();
158  }
159 
160  if (!get_pointer_to_manager_kernel()) {
161  logger::out(logger::level::error, __FILE__, __LINE__, "nullptr: cannot access to manager kernel");
162  throw std::bad_alloc();
163  }
164  auto manager_kernel = *get_pointer_to_manager_kernel();
165  if (!manager_kernel) {
166  logger::out(logger::level::error, __FILE__, __LINE__, "nullptr: cannot access to manager kernel");
167  throw std::bad_alloc();
168  }
169 
170  auto addr = pointer(static_cast<value_type *>(manager_kernel->allocate(n * sizeof(T))));
171  if (!addr) {

178  void priv_deallocate(pointer ptr, [[maybe_unused]] const size_type size) const noexcept {
179  if (!get_pointer_to_manager_kernel()) {

181  return;
182  }
183  auto manager_kernel = *get_pointer_to_manager_kernel();
184  if (!manager_kernel) {
185  logger::out(logger::level::error, __FILE__, __LINE__, "nullptr: cannot access to manager kernel");
186  return;
187  }
188  manager_kernel->deallocate(to_raw_pointer(ptr));
189  }
