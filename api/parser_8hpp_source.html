29 inline bool parse(const bj::value &input_value, value<allocator_type> *out_value) {
30  if (!out_value)return false;
31 
32  if (input_value.is_null()) {
33  out_value->emplace_null();
34  } else if (input_value.is_bool()) {
35  out_value->emplace_bool() = input_value.as_bool();
36  } else if (input_value.is_int64()) {
37  out_value->emplace_int64() = input_value.as_int64();
38  } else if (input_value.is_uint64()) {
39  out_value->emplace_uint64() = input_value.as_uint64();
40  } else if (input_value.is_double()) {
41  out_value->emplace_double() = input_value.as_double();
42  } else if (input_value.is_string()) {
43  out_value->emplace_string() = input_value.as_string().c_str();
44  } else if (input_value.is_array()) {
45  auto &out_array = out_value->emplace_array();
46  for (const auto &item : input_value.as_array()) {
47  out_array.resize(out_array.size() + 1);
48  parse(item, &out_array[out_array.size() - 1]);
49  }
50  } else if (input_value.is_object()) {
51  auto &out_obj = out_value->emplace_object();
52  for (const auto &pair : input_value.as_object()) {
53  auto &val = out_obj[pair.key_c_str()];
54  parse(pair.value(), &val);
55  }
56  } else {
57  std::cerr << "Unexpected Boost::json::kind" << std::endl;
58  std::abort();
59  }
60 
61  return true;
62 }
