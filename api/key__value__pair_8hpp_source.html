248  bool priv_allocate_key(const char_type *const key, const size_type length,
249  char_allocator_type alloc) {
250  assert(m_key_length == 0);
251  m_key_length = length;
252 
253  if (priv_short_key()) {
254  std::char_traits<char_type>::copy(m_short_key, key, m_key_length);
255  std::char_traits<char_type>::assign(m_short_key[m_key_length], '\0');
256  } else {
257  m_long_key = std::allocator_traits<char_allocator_type>::allocate(
258  alloc, m_key_length + 1);
259  if (!m_long_key) {
260  m_key_length = 0;
261  std::abort(); // TODO: change
262  return false;
263  }
264 
265  std::char_traits<char_type>::copy(metall::to_raw_pointer(m_long_key), key,
266  m_key_length);
267  std::char_traits<char_type>::assign(m_long_key[m_key_length], '\0');
268  }
269 
270  return true;
271  }

273  bool priv_deallocate_key(char_allocator_type alloc) {
274  if (m_key_length > k_short_key_max_length) {
275  std::allocator_traits<char_allocator_type>::deallocate(alloc, m_long_key,
276  m_key_length + 1);
277  m_long_key = nullptr;
278  }
279  m_key_length = 0;
280  return true;
