214  bool priv_allocate_key(const char_type *const key, const size_type length) {
215  assert(m_key_length == 0);
216  m_key_length = length;
217 
218  if (m_key_length <= k_short_key_max_length) {
219  std::char_traits<char_type>::copy(m_short_key, key, m_key_length);
220  std::char_traits<char_type>::assign(m_short_key[m_key_length], '\0');
221  } else {
222  char_allocator_type alloc(m_value.get_allocator());
223  m_long_key = std::allocator_traits<char_allocator_type>::allocate(alloc, m_key_length + 1);
224  if (!m_long_key) {
225  m_key_length = 0;
226  std::abort(); // TODO: change
227  return false;
228  }
229 
230  std::char_traits<char_type>::copy(metall::to_raw_pointer(m_long_key), key, m_key_length);
231  std::char_traits<char_type>::assign(m_long_key[m_key_length], '\0');
232  }
233 
234  return true;
235  }

237  bool priv_deallocate_key() {
238  if (m_key_length > k_short_key_max_length) {
239  char_allocator_type alloc(m_value.get_allocator());
240  std::allocator_traits<char_allocator_type>::deallocate(alloc, m_long_key, m_key_length + 1);
241  }
242 
243  m_key_length = 0;
244 
245  return true;
246  }
