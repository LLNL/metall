171  assert(m_key_length == 0);
172  m_key_length = length;
173 
174  if (m_key_length <= k_short_key_max_length) {
175  std::char_traits<char_type>::copy(m_short_key, key, m_key_length);
176  std::char_traits<char_type>::assign(m_short_key[m_key_length], '\0');
177  } else {
178  char_allocator_type alloc(m_value.get_allocator());
179  m_long_key = std::allocator_traits<char_allocator_type>::allocate(alloc, m_key_length + 1);
