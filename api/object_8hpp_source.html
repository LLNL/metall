161  value_postion_type priv_locate_value(std::string_view key) const {
162  auto range = m_index_table.equal_range(hash_key(key));
163  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
164  const auto value_pos = itr->second;
165  if (m_value_storage[value_pos].key() == key) {
166  return value_pos; // Found the key
167  }
168  }
169  return m_value_storage.max_size(); // Couldn't find
170  }

172  value_postion_type priv_emplace_value(std::string_view key, mapped_type mapped_value) {
173  m_value_storage.emplace_back(key, std::move(mapped_value));
174  m_index_table.emplace(hash_key(key), m_value_storage.size() - 1);

182  auto priv_erase(const_iterator value_position) {
183  if (value_position == m_value_storage.cend()) {
184  return m_value_storage.end();
185  }
186 
187  const auto value_position_raw = std::distance(m_value_storage.cbegin(), value_position);
188  bool erased = false;
189  auto range = m_index_table.equal_range(hash_key(value_position->key()));
190  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
191  const auto value_pos = itr->second;
192  if ((ssize_t)value_pos == value_position_raw) {
193  m_index_table.erase(itr);
194  erased = true;
195  break;
196  }
197  }
198  assert(erased);
199 
200  // Update the positions of the values that will be moved forward.
201  for (auto &elem : m_index_table) {
202  if ((ssize_t)elem.second > value_position_raw) {
203  --elem.second;
204  }
205  }
206 
207  // Finally, erase the value.
208  return m_value_storage.erase(value_position);
209  }
