169  static bool remove(const char *root_dir_prefix, const MPI_Comm &comm = MPI_COMM_WORLD) {
170  const int rank = priv_mpi_comm_rank(comm);
171  const int size = priv_mpi_comm_size(comm);
172 
173  // ----- Check if this is a Metall datastore ----- //
174  bool corrent_dir = true;
175  if (!metall::mtlldetail::file_exist(
176  ds::make_root_dir_path(root_dir_prefix) + "/" + k_datastore_mark_file_name)) {
177  corrent_dir = false;
178  }
179  if (!priv_global_and(corrent_dir, comm)) {
180  return false;
181  }
182 
183  // ----- Check if #of MPI processes matches ----- //
184  bool correct_mpi_size = true;
185  if (rank == 0) {
186  const int read_size = priv_read_num_partitions(root_dir_prefix, comm);
187  if (read_size != size) {
188  correct_mpi_size = false;
189  std::stringstream ss;
190  ss << " Invalid number of MPI processes (provided " << size << ", " << "expected " << correct_mpi_size << ")";
191  logger::out(logger::level::error, __FILE__, __LINE__, ss.str().c_str());
192  }
193  }
194  if (!priv_global_and(correct_mpi_size, comm)) {
195  return false;
196  }
197 
198  // ----- Remove directories ----- //
199  bool ret = true;
200  for (int i = 0; i < size; ++i) {
201  if (i == rank) {
202  if (metall::mtlldetail::file_exist(ds::make_root_dir_path(root_dir_prefix))
203  && !metall::mtlldetail::remove_file(ds::make_root_dir_path(root_dir_prefix))) {
204  std::string s("Failed to remove directory: " + ds::make_root_dir_path(root_dir_prefix));
205  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
206  ret = false;
207  }
208  }
209  priv_mpi_barrier(comm);
210  }
211 
212  return priv_global_and(ret, comm);
213  }

242  static void priv_setup_root_dir(const std::string &root_dir_prefix, const MPI_Comm &comm) {
243  const int rank = priv_mpi_comm_rank(comm);
244  const int size = priv_mpi_comm_size(comm);
245  const std::string root_dir_path = ds::make_root_dir_path(root_dir_prefix);
246 
247  // Make sure the root directory and a file with the same name do not exist
248  const auto local_ret = metall::mtlldetail::file_exist(root_dir_path);
249  if (priv_global_or(local_ret, comm)) {
250  if (rank == 0) {
251  std::string s("Root directory (or a file with the same name) already exists: " + root_dir_path);
252  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
253  ::MPI_Abort(comm, -1);
254  }
255  }
256  priv_mpi_barrier(comm);
257 
258  for (int i = 0; i < size; ++i) {
259  if (i == rank && !metall::mtlldetail::directory_exist(root_dir_path)) {
260  if (!metall::mtlldetail::create_directory(root_dir_path)) {
261  std::string s("Failed to create directory: " + root_dir_path);
262  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
263  ::MPI_Abort(comm, -1);
264  }
265 
266  const std::string mark_file = root_dir_path + "/" + k_datastore_mark_file_name;
267  if (!metall::mtlldetail::create_file(mark_file)) {
268  std::string s("Failed to create file: " + mark_file);
269  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
270  ::MPI_Abort(comm, -1);
271  }
272 
273  priv_store_partition_size(root_dir_prefix, comm);
274  }
275  priv_mpi_barrier(comm);
276  }

279  static void priv_store_partition_size(const std::string &root_dir_prefix, const MPI_Comm &comm) {
280  const int size = priv_mpi_comm_size(comm);
281  const std::string path = ds::make_root_dir_path(root_dir_prefix) + "/" + k_partition_size_file_name;
282 
283  std::ofstream ofs(path);
284  if (!ofs) {
285  std::string s("Failed to create a file: " + path);
286  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
287  ::MPI_Abort(comm, -1);
288  }
289  ofs << size;
290  if (!ofs) {
291  std::string s("Failed to write data to a file: " + path);
292  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
293  ::MPI_Abort(comm, -1);
294  }
295  ofs.close();
296  }

298  static int priv_read_num_partitions(const std::string &root_dir_prefix, const MPI_Comm &comm) {
299  const std::string path = ds::make_root_dir_path(root_dir_prefix) + "/" + k_partition_size_file_name;
300  std::ifstream ifs(path);
301  if (!ifs) {
302  std::string s("Failed to open a file: " + path);
303  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
304  ::MPI_Abort(comm, -1);
305  }
306  int read_size = -1;
307  if (!(ifs >> read_size)) {
308  std::string s("Failed to read data from: " + path);
309  logger::out(logger::level::error, __FILE__, __LINE__, s.c_str());
310  ::MPI_Abort(comm, -1);
311  }
312 
313  return read_size;
314  }
