30 class SplitMix64 {
31  public:
32 
33  explicit SplitMix64(const uint64_t seed)
34  : m_x(seed) {}
35 
36  // This is a fixed-increment version of Java 8's SplittableRandom generator
37  // See http://dx.doi.org/10.1145/2714064.2660195 and
38  // http://docs.oracle.com/javase/8/docs/api/java/util/SplittableRandom.html
39  //
40  // It is a very fast generator passing BigCrush, and it can be useful if
41  // for some reason you absolutely want 64 bits of state.
42  uint64_t next() {
43  uint64_t z = (m_x += 0x9e3779b97f4a7c15);
44  z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9;
45  z = (z ^ (z >> 27)) * 0x94d049bb133111eb;
46  return z ^ (z >> 31);
47  }
48 
49  private:
50  uint64_t m_x; /* The state can be seeded with any value. */
51 };

65 // This is xoshiro512++ 1.0, one of our all-purpose, rock-solid
66 // generators. It has excellent (about 1ns) speed, a state (512 bits) that
67 // is large enough for any parallel application, and it passes all tests
68 // we are aware of.
69 //
70 // For generating just floating-point numbers, xoshiro512+ is even faster.
71 //
72 // The state must be seeded so that it is not everywhere zero. If you have
73 // a 64-bit seed, we suggest to seed a splitmix64 generator and use its
74 // output to fill s.
75 // -----------------------------------------------------------------------------
76 class xoshiro512pp {
77  public:
78  using result_type = uint64_t;
79 
80  explicit xoshiro512pp(const uint64_t seed) {
81  SplitMix64 seed_gen(seed);
82  for (int i = 0; i < 8; ++i) {
83  m_s[i] = seed_gen.next();
84  }
85  }
86 
87  bool equal(const xoshiro512pp &other) const {
88  for (int i = 0; i < 8; ++i) {
89  if (m_s[i] != other.m_s[i]) return false;
90  }
91  return true;
92  }
93 
94  result_type next() {
95  const uint64_t result = rotl(m_s[0] + m_s[2], 17) + m_s[2];
96 
97  const uint64_t t = m_s[1] << 11;
98 
99  m_s[2] ^= m_s[0];
100  m_s[5] ^= m_s[1];
101  m_s[1] ^= m_s[2];
102  m_s[7] ^= m_s[3];
103  m_s[3] ^= m_s[4];
104  m_s[4] ^= m_s[5];
105  m_s[0] ^= m_s[6];
106  m_s[6] ^= m_s[7];
107 
108  m_s[6] ^= t;
109 
110  m_s[7] = rotl(m_s[7], 21);
111 
112  return result;
113  }

115  // This is the jump function for the generator. It is equivalent
116  // to 2^256 calls to next(); it can be used to generate 2^256
117  // non-overlapping subsequences for parallel computations.
118  void jump() {
119  static const uint64_t
120  JUMP[] = {0x33ed89b6e7a353f9, 0x760083d7955323be, 0x2837f2fbb5f22fae, 0x4b8c5674d309511c, 0xb11ac47a7ba28c25,
121  0xf1be7667092bcc1c, 0x53851efdb6df0aaf, 0x1ebbc8b23eaf25db};
122 
123  uint64_t t[sizeof m_s / sizeof *m_s];
124  memset(t, 0, sizeof t);
125  for (int i = 0; i < (int)(sizeof JUMP / sizeof *JUMP); i++)
126  for (int b = 0; b < 64; b++) {
127  if (JUMP[i] & UINT64_C(1) << b)
128  for (int w = 0; w < (int)(sizeof m_s / sizeof *m_s); w++)
129  t[w] ^= m_s[w];
130  next();
131  }
132 
133  memcpy(m_s, t, sizeof m_s);
134  }
135 
136  // This is the long-jump function for the generator. It is equivalent to
137  // 2^384 calls to next(); it can be used to generate 2^128 starting points,
138  // from each of which jump() will generate 2^128 non-overlapping
139  // subsequences for parallel distributed computations.
140  void long_jump() {
141  static const uint64_t LONG_JUMP[] = {0x11467fef8f921d28, 0xa2a819f2e79c8ea8, 0xa8299fc284b3959a, 0xb4d347340ca63ee1,
142  0x1cb0940bedbff6ce, 0xd956c5c4fa1f8e17, 0x915e38fd4eda93bc,
143  0x5b3ccdfa5d7daca5};
144 
145  uint64_t t[sizeof m_s / sizeof *m_s];
146  memset(t, 0, sizeof t);
147  for (int i = 0; i < (int)(sizeof LONG_JUMP / sizeof *LONG_JUMP); i++)
148  for (int b = 0; b < 64; b++) {
149  if (LONG_JUMP[i] & UINT64_C(1) << b)
150  for (int w = 0; w < (int)(sizeof m_s / sizeof *m_s); w++)
151  t[w] ^= m_s[w];
152  next();
153  }
154 
155  memcpy(m_s, t, sizeof m_s);
156  }
157 
158  private:
159  static constexpr uint64_t rotl(const uint64_t x, int k) {
160  return (x << k) | (x >> (64 - k));
161  }
162 
163  uint64_t m_s[8];
164 };

178 // This is xoroshiro1024++ 1.0, one of our all-purpose, rock-solid,
179 // large-state generators. It is extremely fast and it passes all
180 // tests we are aware of. Its state however is too large--in general,
181 // the xoshiro256 family should be preferred.
182 //
183 // For generating just floating-point numbers, xoroshiro1024* is even
184 // faster (but it has a very mild bias, see notes in the comments).
185 //
186 // The state must be seeded so that it is not everywhere zero. If you have
187 // a 64-bit seed, we suggest to seed a splitmix64 generator and use its
188 // output to fill s.
189 // -----------------------------------------------------------------------------
190 class xoshiro1024pp {
191  public:
192  using result_type = uint64_t;
193 
194  explicit xoshiro1024pp(const uint64_t seed) {
195  SplitMix64 seed_gen(seed);
196  for (int i = 0; i < 16; ++i) {
197  m_s[i] = seed_gen.next();
198  }
199  }
200 
201  bool equal(const xoshiro1024pp &other) const {
202  for (int i = 0; i < 16; ++i) {
203  if (m_s[i] != other.m_s[i]) return false;
204  }
205  return true;
206  }
207 
208  uint64_t next() {
209  const int q = m_p;
210  const uint64_t s0 = m_s[m_p = (m_p + 1) & 15];
211  uint64_t s15 = m_s[q];
212  const uint64_t result = rotl(s0 + s15, 23) + s15;
213 
214  s15 ^= s0;
215  m_s[q] = rotl(s0, 25) ^ s15 ^ (s15 << 27);
216  m_s[m_p] = rotl(s15, 36);
217 
218  return result;
219  }

226  0xb59422e0b9138c5f, 0xf06a6afb49d668bb, 0xacb8a6412c8a1401,

230  0x45f387286ade3205, 0x03864f454a8920bd, 0xa68fa28725b1b384};

233  memset(t, 0, sizeof t);
234  for (int i = 0; i < (int)(sizeof JUMP / sizeof *JUMP); i++)
235  for (int b = 0; b < 64; b++) {
236  if (JUMP[i] & UINT64_C(1) << b)
237  for (int j = 0; j < int(sizeof m_s / sizeof *m_s); j++)

244  }
245  }
