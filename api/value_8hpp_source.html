270  os << val.as_int64() << "";
271  } else if (val.is_uint64()) {
272  os << val.as_uint64() << "";
273  } else if (val.is_double()) {
274  os << val.as_double() << "";
275  } else if (val.is_string()) {
276  os << "\"" << val.as_string() << "\"";
277  } else if (val.is_array()) {
278  os << val.as_array() << "";
279  } else if (val.is_object()) {

281  } else {
