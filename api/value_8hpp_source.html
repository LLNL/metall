26 template <typename allocator_type, typename other_value_type>
27 inline bool general_value_equal(const value<allocator_type> &value, const other_value_type &other_value) noexcept {
28  if (other_value.is_null()) {
29  return value.is_null();
30  } else if (other_value.is_bool()) {
31  return value.is_bool() && value.as_bool() == other_value.as_bool();
32  } else if (other_value.is_int64()) {
33  if (value.is_int64()) {
34  return value.as_int64() == other_value.as_int64();
35  }
36  if (value.is_uint64()) {
37  return (other_value.as_int64() < 0) ? false : value.as_uint64()
38  == static_cast<std::uint64_t>(other_value.as_int64());
39  }
40  return false;
41  } else if (other_value.is_uint64()) {
42  if (value.is_uint64()) {
43  return value.as_uint64() == other_value.as_uint64();
44  }
45  if (value.is_int64()) {
46  return (value.as_int64() < 0) ? false : static_cast<std::uint64_t>(value.as_int64()) == other_value.as_uint64();
47  }
48  return false;
49  } else if (other_value.is_double()) {
50  return value.is_double() && value.as_double() == other_value.as_double();
51  } else if (other_value.is_object()) {
52  return value.is_object() && (value.as_object() == other_value.as_object());
53  } else if (other_value.is_array()) {
54  return value.is_array() && (value.as_array() == other_value.as_array());
55  } else if (other_value.is_string()) {
56  if (!value.is_string()) return false;
57  const auto &str = value.as_string();
58  const auto &other_srt = other_value.as_string();
59  return str.compare(other_srt.c_str()) == 0;
60  }
61 
62  assert(false);
63  return false;
64 }
