26 template <typename allocator_type, typename other_value_type>
27 inline bool general_value_equal(const value<allocator_type>& value, const other_value_type& other_value) noexcept {
28  if (other_value.is_null()) {
29  return value.is_null();
30  } else if (other_value.is_bool()) {
31  return value.is_bool() && value.as_bool() == other_value.as_bool();
32  } else if (other_value.is_int64()) {
33  if (value.is_int64()) {
34  return value.as_int64() == other_value.as_int64();
35  }
36  if (value.is_uint64()) {
37  return (other_value.as_int64() < 0) ? false : value.as_uint64() == static_cast<std::uint64_t>(other_value.as_int64());
38  }
39  return false;
40  } else if (other_value.is_uint64()) {
41  if (value.is_uint64()) {
42  return value.as_uint64() == other_value.as_uint64();
43  }
44  if (value.is_int64()) {
45  return (value.as_int64() < 0) ? false : static_cast<std::uint64_t>(value.as_int64()) == other_value.as_uint64();
46  }
47  return false;
48  } else if (other_value.is_double()) {
49  return value.is_double() && value.as_double() == other_value.as_double();
50  } else if (other_value.is_object()) {
51  return value.is_object() && (value.as_object() == other_value.as_object());
52  } else if (other_value.is_array()) {
53  return value.is_array() && (value.as_array() == other_value.as_array());
54  } else if (other_value.is_string()) {
55  return value.is_string() && (value.as_string() == other_value.as_string());
56  }
57 
58  assert(false);
59  return false;
60 }
