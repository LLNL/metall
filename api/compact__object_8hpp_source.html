30 template <typename allocator_type, typename other_object_type>
31 inline bool general_compact_object_equal(const compact_object<allocator_type> &object,
32  const other_object_type &other_object) noexcept {
33  if (object.size() != other_object.size()) return false;
34 
35  for (const auto &key_value: object) {
36  auto itr = other_object.find(key_value.key_c_str());
37  if (itr == other_object.end()) return false;
38  if (key_value.value() != itr->value()) return false;
39  }
40 
41  return true;
42 }
