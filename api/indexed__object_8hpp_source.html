201  value_postion_type priv_locate_value(const key_type &key) const {
202  auto range = m_index_table.equal_range(hash_key(key));
203  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
204  const auto value_pos = itr->second;
205  if (m_value_storage[value_pos].key() == key) {
206  return value_pos; // Found the key
207  }
208  }
209  return m_value_storage.max_size(); // Couldn't find
210  }
211 
212  value_postion_type priv_emplace_value(const key_type &key, mapped_type mapped_value) {
213  m_value_storage.emplace_back(key, std::move(mapped_value));
214  m_index_table.emplace(hash_key(key), m_value_storage.size() - 1);
215  return m_value_storage.size() - 1;
216  }
217 
218  static auto hash_key(const key_type &key) {
219  return metall::mtlldetail::MurmurHash64A(key.data(), key.length(), 123);
220  }

222  auto priv_erase(const_iterator value_position) {
223  if (value_position == m_value_storage.cend()) {
224  return m_value_storage.end();
225  }
226 
227  const auto value_position_raw = std::distance(m_value_storage.cbegin(), value_position);
228  [[maybe_unused]] bool erased = false;
229  auto range = m_index_table.equal_range(hash_key(value_position->key()));
230  for (auto itr = range.first, end = range.second; itr != end; ++itr) {
231  const auto value_pos = itr->second;
232  if ((ssize_t)value_pos == value_position_raw) {
233  m_index_table.erase(itr);
234  erased = true;
235  break;
236  }
237  }
238  assert(erased);
239 
240  // Update the positions of the values that will be moved forward.
241  for (auto &elem : m_index_table) {
242  if ((ssize_t)elem.second > value_position_raw) {
243  --elem.second;
244  }
245  }
246 
247  // Finally, erase the value.
248  return m_value_storage.erase(value_position);
249  }
