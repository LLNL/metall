20 template <typename allocator_type>
21 inline void value_to_impl_helper(const mj::value<allocator_type> &input_value, bj::value *out_bj_value) {
22  if (input_value.is_bool()) {
23  *out_bj_value = input_value.as_bool();
24  } else if (input_value.is_int64()) {
25  *out_bj_value = input_value.as_int64();
26  } else if (input_value.is_uint64()) {
27  *out_bj_value = input_value.as_uint64();
28  } else if (input_value.is_double()) {
29  *out_bj_value = input_value.as_double();
30  } else if (input_value.is_string()) {
31  *out_bj_value = input_value.as_string();
32  } else if (input_value.is_array()) {
33  bj::array bj_array;
34  for (const auto &elem : input_value.as_array()) {
35  bj_array.emplace_back(mj::value_to<bj::value>(elem));
36  }
37  *out_bj_value = bj_array;
38  } else if (input_value.is_object()) {
39  bj::object bj_object;
40  for (const auto &elem : input_value.as_object()) {
41  bj_object[elem.key().data()] = mj::value_to<bj::value>(elem.value());
42  }
43  *out_bj_value = bj_object;
44  } else if (input_value.is_null()) {
45  out_bj_value->emplace_null();
46  }
47 }
